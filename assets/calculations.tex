%Calculations
\FPeval{\ddbStrength}{trunc((\ddStrength -10) /2:0)}
\FPeval{\ddbDexterity}{trunc((\ddDexterity -10) /2:0)}
\FPeval{\ddbConstitution}{trunc((\ddConstitution -10) /2:0)}
\FPeval{\ddbIntelligence}{trunc((\ddIntelligence -10) /2:0)}
\FPeval{\ddbWisdom}{trunc((\ddWisdom -10) /2:0)}
\FPeval{\ddbCharisma}{trunc((\ddCharisma -10) /2:0)}
